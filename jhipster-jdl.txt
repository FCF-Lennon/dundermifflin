entity Departamento {
    nombreDepartamento String required,
    ubicacionDepartamento String required
}

entity Jefe {
    nombreJefe String required
}

entity InformacionContactoJefe {
    telefonoJefe String required,
    tipoTelefono String required
}

entity Empleado {
    nombreEmpleado String required
    apellidoEmpleado String required,
    correo String required
}

entity InformacionContactoEmpleado {
    telefono String required,
    tipoTelefono String required
}

entity EmpleadoDepartamento {
    fechaAsignacion ZonedDateTime
}

entity Presupuesto {
    presupuestoDepartamento BigDecimal required
}

relationship OneToOne {
    Jefe {InformacionContactoJefe} to InformacionContactoJefe,
    Empleado{InformacionContactoEmpleado} to InformacionContactoEmpleado
}

relationship OneToMany {
    Departamento{Presupuesto} to Presupuesto
    Departamento{EmpleadoDepartamento} to EmpleadoDepartamento,
    Jefe{Empleado} to Empleado
}

relationship ManyToOne {
    EmpleadoDepartamento{Departamento} to Departamento,
    EmpleadoDepartamento{Empleado} to Empleado
}

relationship ManyToMany {
    Departamento{Jefe} to Jefe
}
